Table PATIENT {
  health_id varchar(50) [pk]
  fname varchar(100) [not null]
  mname varchar(100)
  lname varchar(100) [not null]

  address_street varchar(255)
  address_city varchar(100)
  address_state varchar(100)
  postal_code varchar(20)

  date_of_birth date [not null]

  gender varchar(1) [not null, note: "ENUM('M','F','O')"]
  blood_group varchar(3) [note: "ENUM('A+','A-','B+','B-','AB+','AB-','O+','O-')"]

  emergency_contact_name varchar(255)
  emergency_contact_phone varchar(20)

  insurance_provider varchar(255)
  insurance_start date
  insurance_end date
  policy_type varchar(100)

  Note: "CHECK: insurance_end IS NULL OR insurance_start IS NULL OR insurance_end >= insurance_start"
}

Table HOSPITAL {
  hospital_id bigint [pk, increment]
  hospital_name varchar(255) [not null]
  license_number varchar(100) [not null, unique]
  phone varchar(20)
  email varchar(100)
  bed_capacity int

  address_street varchar(255)
  address_city varchar(100)
  address_state varchar(100)
  postal_code varchar(20)

  Note: "CHECK: bed_capacity IS NULL OR bed_capacity >= 0"
}

Table DOCTOR {
  doctor_id bigint [pk, increment]
  medical_license_number varchar(100) [not null, unique]
  name varchar(255) [not null]
  phone_number varchar(20)
  email varchar(100)
  date_of_birth date
}

Table MEDICATION {
  medication_id bigint [pk, increment]
  generic_name varchar(255) [not null]
  brand_name varchar(255)
  drug_class varchar(100)
  manufacturer varchar(255)

  Indexes {
    (generic_name, brand_name) [unique, name: "uq_medication"]
  }
}

Table LAB_TEST_CATALOG {
  test_code varchar(50) [pk]
  test_name varchar(255) [not null]
  specimen_type varchar(100)
  collection_procedure text
}

Table DIAGNOSIS_CODE {
  icd10_code varchar(20) [pk]
  description text [not null]
  category varchar(100)
}

Table PROCEDURE_CODE {
  cpt_code varchar(20) [pk]
  description text [not null]
  category varchar(100)
  base_cost decimal(10,2)

  Note: "CHECK: base_cost IS NULL OR base_cost >= 0"
}

Table ENCOUNTER {
  encounter_id bigint [pk, increment]

  health_id varchar(50) [not null]
  hospital_id bigint

  encounter_date_time datetime [not null]
  encounter_type varchar(20) [not null, note: "ENUM('Outpatient','Inpatient','Emergency','Follow-up')"]

  admission_date_time datetime
  discharge_date_time datetime
  bed_number varchar(20)

  chief_complaint text
  examination_notes text
  treatment_plan text

  Note: '''
  CHECK: discharge_date_time IS NULL OR admission_date_time IS NULL OR discharge_date_time >= admission_date_time
  CHECK: encounter_type <> 'Inpatient' OR admission_date_time IS NOT NULL
  CHECK: encounter_type <> 'Inpatient' OR bed_number IS NOT NULL
  '''

}

Table ENCOUNTER_DOCTOR {
  encounter_id bigint [not null]
  doctor_id bigint [not null]
  role varchar(50)
  assigned_time datetime
  is_primary boolean [default: false]

  Indexes {
    (encounter_id, doctor_id) [pk]
  }
}

Table PRESCRIPTION {
  prescription_id bigint [pk, increment]
  encounter_id bigint [not null]
  doctor_id bigint [not null]

  prescription_date date
  start_date date
  end_date date

  status varchar(20) [default: "Active", note: "ENUM('Active','Completed','Discontinued','Cancelled')"]

  Note: "CHECK: end_date IS NULL OR start_date IS NULL OR end_date >= start_date"
}

Table PRESCRIPTION_ITEM {
  prescription_id bigint [not null]
  medication_id bigint [not null]

  dosage_strength varchar(50)
  dosage_form varchar(50)
  frequency varchar(100)

  duration_days int
  quantity_dispensed int

  instructions text

  Indexes {
    (prescription_id, medication_id) [pk]
  }

  Note: '''
  CHECK: duration_days IS NULL OR duration_days >= 0
  CHECK: quantity_dispensed IS NULL OR quantity_dispensed >= 0
  '''

}

Table LAB_ORDER {
  lab_order_id bigint [pk, increment]
  encounter_id bigint [not null]
  doctor_id bigint [not null]
  test_code varchar(50) [not null]

  order_date_time datetime [not null, default: `CURRENT_TIMESTAMP`]
  priority varchar(10) [not null, note: "ENUM('Routine','Urgent','Stat')"]
  clinical_info text
  order_status varchar(20) [default: "Pending", note: "ENUM('Pending','In Progress','Completed','Cancelled')"]
}

Table VITAL_SIGNS {
  encounter_id bigint [not null]
  reading_timestamp datetime [not null]

  bp_systolic int
  bp_diastolic int
  pulse int
  temperature decimal(4,1)
  respiratory_rate int
  height decimal(5,2)
  weight decimal(5,2)
  oxygen_saturation int

  Indexes {
    (encounter_id, reading_timestamp) [pk]
  }

  Note: "Checks: bp_systolic 50-250, bp_diastolic 30-150, pulse 20-250, temp 30-45, rr 5-60, height 30-250, weight 1-500, oxygen 50-100 (all nullable)"
}

Table LAB_RESULT {
  result_id bigint [pk, increment]
  lab_order_id bigint [not null]

  result_value varchar(100)
  result_unit varchar(50)
  reference_range varchar(100)

  abnormal_flag boolean [default: false]
  critical_flag boolean [default: false]

  result_date datetime
  verified_by_doctor_id bigint

  Note: "CHECK: NOT critical_flag OR abnormal_flag"
}

Table ENCOUNTER_DIAGNOSIS {
  encounter_id bigint [not null]
  icd10_code varchar(20) [not null]

  diagnosis_type varchar(20) [not null, note: "ENUM('Primary','Secondary','Provisional')"]
  status varchar(20) [not null, note: "ENUM('Active','Resolved','Ruled Out')"]
  diagnosed_date date

  Indexes {
    (encounter_id, icd10_code) [pk]
  }
}

Table ENCOUNTER_PROCEDURE {
  encounter_id bigint [not null]
  cpt_code varchar(20) [not null]
  sequence_number int [not null]

  procedure_date date
  procedure_notes text
  complications text

  Indexes {
    (encounter_id, cpt_code, sequence_number) [pk]
  }

  Note: "CHECK: sequence_number >= 1"
}

Table DRUG_INTERACTION {
  medication_id_1 bigint [not null]
  medication_id_2 bigint [not null]

  interaction_description text
  severity varchar(20) [not null, note: "ENUM('Mild','Moderate','Severe','Life-threatening')"]
  recommendation text

  Indexes {
    (medication_id_1, medication_id_2) [pk]
  }

  Note: "CHECK: medication_id_1 < medication_id_2 (canonical ordering)"
}

Table DOCTOR_HOSPITAL {
  doctor_id bigint [not null]
  hospital_id bigint [not null]

  join_date date [not null]
  end_date date
  status varchar(20) [default: "Active", note: "ENUM('Active','Inactive','On Leave')"]

  Indexes {
    (doctor_id, hospital_id) [pk]
  }

  Note: "CHECK: end_date IS NULL OR end_date >= join_date"
}

Table CONSENT {
  consent_id bigint [pk, increment]

  health_id varchar(50) [not null]
  hospital_id bigint [not null]

  access_level varchar(20) [not null, note: "ENUM('Basic','Restricted','Full')"]
  purpose text

  effective_date date [not null]
  expiration_date date
  revoked_date date

  status varchar(20) [default: "Active", note: "ENUM('Active','Expired','Revoked')"]
  created_timestamp datetime [not null, default: `CURRENT_TIMESTAMP`]

  Note: '''
  CHECK: expiration_date IS NULL OR expiration_date >= effective_date
  CHECK: revoked_date IS NULL OR revoked_date >= effective_date
  CHECK: status <> 'Revoked' OR revoked_date IS NOT NULL
  '''

}

  status varchar(20) [default: "Active", note: "ENUM('Active','Inactive','Locked')"]
  created_date datetime [not null, default: `CURRENT_TIMESTAMP`]
  last_login datetime
}
